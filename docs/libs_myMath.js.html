/**
 * My Math library
 * @module MyMath
 */

/**
 * Add two numbers
 * @param {number} x First Number
 * @param {number} y Second Number
 * @returns {number} Sum of X and Y
 */
const add = (x, y) => x + y

/**
 * Substract Two numbers
 * @param {number} x First Number
 * @param {number} y Second Number
 * @returns {number} substract of X and Y
 */
const substract = (x, y) => x - y

module.exports = {
  add,
  substract
}
